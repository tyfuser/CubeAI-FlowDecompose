# ========================================
# 魔方 AI - 环境变量配置示例
# ========================================
# 
# 使用说明：
# 1. 复制此文件内容创建 .env 文件
# 2. 根据实际环境填写相应的值
# 3. 不要将 .env 文件提交到版本控制

# ========================================
# API 配置
# ========================================

# API 基础地址
# 开发环境 - Dashboard API（端口8000）
# 注意：如果前端使用HTTPS，但后端是HTTP，需要明确设置HTTP URL
# 例如：VITE_API_BASE_URL=http://192.168.43.226:8000/api/v1
VITE_API_BASE_URL=http://localhost:8000/api/v1

# 视频分析API地址（端口8000，与Dashboard API相同）
# 如果后端不支持HTTPS，使用HTTP（会有混合内容警告，但能工作）
# 例如：VITE_SHOT_ANALYSIS_BASE_URL=http://192.168.43.226:8000
VITE_SHOT_ANALYSIS_BASE_URL=http://localhost:8000

# Phone AI后端端口（端口8001，通常使用HTTPS）
# VITE_PHONE_AI_PORT=8001

# 测试环境
# VITE_API_BASE_URL=https://test-api.rubik-ai.com/v1

# 生产环境
# VITE_API_BASE_URL=https://api.rubik-ai.com/v1

# API 超时时间（毫秒）
VITE_API_TIMEOUT=30000

# ========================================
# 认证配置
# ========================================

# JWT Token 存储 Key
VITE_TOKEN_KEY=rubik_token

# Refresh Token 存储 Key
VITE_REFRESH_TOKEN_KEY=rubik_refresh_token

# ========================================
# 第三方服务配置
# ========================================

# Google Gemini API Key（如果仍需要直接调用）
VITE_GEMINI_API_KEY=your_gemini_api_key_here

# ========================================
# 功能开关
# ========================================

# 是否启用 Mock 数据（开发阶段可设为 true）
VITE_ENABLE_MOCK=false

# 是否启用 API 日志
VITE_ENABLE_API_LOG=true

# ========================================
# 业务配置
# ========================================

# 每日免费分析配额
VITE_FREE_DAILY_QUOTA=5

# 文件上传大小限制（MB）
VITE_MAX_FILE_SIZE=100

# 支持的视频格式
VITE_SUPPORTED_VIDEO_FORMATS=mp4,mov,avi,mkv

# 轮询间隔（毫秒）
VITE_POLL_INTERVAL=2000

# 最大轮询次数
VITE_MAX_POLL_ATTEMPTS=60
